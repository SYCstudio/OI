#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
#include<vector>
#include<map>
using namespace std;

int n,m;
map<int,int> Map;
vector<int> Arr;

int main()
{
    int x;
    cin>>n>>m;
    for (int i=1;i<=n;i++)
    {
        scanf("%d",&x);
        Map[x]=i;
    }
    for (int i=1;i<=m;i++)//将LCS问题转换为LIS问题，降低时间复杂度
    {
        scanf("%d",&x);
        x=Map[x];
        if (x==0)
            continue;
        if ((Arr.size()==0)||(x>Arr[Arr.size()-1]))
            Arr.push_back(x);
        else
            *lower_bound(Arr.begin(),Arr.end(),x)=x;
    }
    cout<<Arr.size()<<endl;
    return 0;
}
